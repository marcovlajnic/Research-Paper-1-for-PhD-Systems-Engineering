# Load data
# Importing original Our World in Data Dataset consisting of 26 countries (24 European Countries plus USA and Canada) into a Pandas Python dataframe
data = pd.read_csv("C:/Users/marco/Downloads/Grouped_data(3).csv", header=0, usecols=['location', 'male_smokers', 'life_expectancy', 'aged_65_older', 'median_age'])
data = data.dropna()

# Separate the country names from the numeric data
countries = data.iloc[:, 0]
numeric_data = data.iloc[:, 1:]

# Scale the numeric data
scaler = StandardScaler()
scaled_data = scaler.fit_transform(numeric_data)

# Determine the optimal number of clusters using the elbow method
wcss = []
for i in range(1, 11):
    kmeans = KMeans(n_clusters=i, init='k-means++', max_iter=300, n_init=10, random_state=0)
    kmeans.fit(scaled_data)
    wcss.append(kmeans.inertia_)

plt.plot(range(1, 11), wcss)
plt.title('Elbow Method')
plt.xlabel('Number of clusters')
plt.ylabel('WCSS')
plt.show()

# Print the optimal number of clusters
optimal_n_clusters = np.argmin(np.diff(wcss)) + 2
print("Optimal number of clusters:", optimal_n_clusters)

# Fit K-means clustering algorithm
kmeans = KMeans(n_clusters=optimal_n_clusters, random_state=42)
kmeans.fit(scaled_data)

# Get cluster labels for each data point
labels = kmeans.labels_

# Print the countries in each cluster
for i in range(optimal_n_clusters):
    print("Countries in cluster ", i+1, ":")
    print(data.iloc[labels == i]['location'].tolist())

# Get the features that are the most different
centroids = kmeans.cluster_centers_
max_differences = np.max(centroids, axis=0) - np.min(centroids, axis=0)
features = list(data.columns[1:])
max_differences_dict = dict(zip(features, max_differences))
print("Features that are the most different across clusters:")
print(max_differences_dict)
